Google Gemini API

Google Gemini는 텍스트, 이미지, 비디오, 오디오를 네이티브로 처리하는 멀티모달 AI 모델입니다. 업계 최장인 2M 토큰 컨텍스트와 강력한 무료 티어를 제공하여 다양한 AI 애플리케이션을 구축할 수 있습니다.

업데이트 안내: 모델/요금/버전/정책 등 시점에 민감한 정보는 변동될 수 있습니다. 최신 내용은 공식 문서를 확인하세요.
핵심 포인트
  • Gemini 1.5 Pro: 2M 토큰 컨텍스트, 멀티모달
  • Gemini 1.5 Flash: 빠른 속도, 저렴한 가격
  • 무료 티어: AI Studio에서 제한적 무료 사용
  • 네이티브 멀티모달: 텍스트, 이미지, 비디오, 오디오
  • Google 서비스 통합: Search, Maps, YouTube

비디오 입력 (Gemini 특화)

Python 
import google.generativeai as genai

genai.configure(api_key="AIzaSy...")

# 1. 비디오 파일 업로드
video_file = genai.upload_file(path="video.mp4")

print(f"업로드 완료: {video_file.uri}")

# 2. 업로드 완료 대기
import time
while video_file.state.name == "PROCESSING":
    time.sleep(2)
    video_file = genai.get_file(video_file.name)

if video_file.state.name == "FAILED":
    raise ValueError("비디오 처리 실패")

# 3. 비디오 분석
model = genai.GenerativeModel("gemini-1.5-pro")

response = model.generate_content([
    video_file,
    "이 비디오의 주요 내용을 요약해주세요. 중요한 장면을 시간대별로 설명해주세요."
])

print(response.text)

# 4. 파일 삭제 (선택사항)
genai.delete_file(video_file.name)

Python 
import os

# 프로젝트의 모든 Python 파일 수집
codebase = ""
for root, dirs, files in os.walk("./src"):
    for file in files:
        if file.endswith(".py"):
            filepath = os.path.join(root, file)
            with open(filepath, "r") as f:
                codebase += f"\n\n{'='*50}\n"
                codebase += f"파일: {filepath}\n"
                codebase += f"{'='*50}\n"
                codebase += f.read()

print(f"코드베이스 크기: {len(codebase)} 문자")

# 전체 코드베이스 분석
model = genai.GenerativeModel("gemini-1.5-pro")

response = model.generate_content([
    f"다음은 Python 프로젝트의 전체 코드입니다:\n\n{codebase}\n\n",

Function Calling

함수 정의

Python 
import google.generativeai as genai

# 함수 정의
get_weather_function = genai.protos.FunctionDeclaration(
    name="get_weather",
    description="특정 위치의 날씨를 가져옵니다",
    parameters=genai.protos.Schema(
        type=genai.protos.Type.OBJECT,
        properties={
            "location": genai.protos.Schema(
                type=genai.protos.Type.STRING,
                description="도시 이름"
            ),
            "unit": genai.protos.Schema(
                type=genai.protos.Type.STRING,
                enum=["celsius", "fahrenheit"]
            )
        },
        required=["location"]
    )
)

# 툴 생성
weather_tool = genai.protos.Tool(
    function_declarations=[get_weather_function]
)

# 모델에 툴 제공
model = genai.GenerativeModel(
    "gemini-1.5-flash",
    tools=[weather_tool]
)

# 함수 호출 유도
response = model.generate_content("서울의 날씨는 어때?")

# 함수 호출 확인
if response.candidates[0].content.parts[0].function_call:
    function_call = response.candidates[0].content.parts[0].function_call
    print(f"함수 이름: {function_call.name}")
    print(f"인자: {dict(function_call.args)}")

함수 실행 루프

Python 
import json

# 실제 함수 구현
def get_weather(location: str, unit: str = "celsius"):
    # 실제 API 호출 대신 더미 데이터
    return {
        "location": location,
        "temperature": 22,
        "unit": unit,
        "condition": "맑음"
    }

# 채팅 세션 시작
chat = model.start_chat()

response = chat.send_message("서울 날씨 알려줘")

# 함수 호출 처리
if response.candidates[0].content.parts[0].function_call:
    function_call = response.candidates[0].content.parts[0].function_call

    # 함수 실행
    result = get_weather(**dict(function_call.args))

    # 결과를 Gemini에게 반환
    response2 = chat.send_message(
        genai.protos.Content(
            parts=[genai.protos.Part(
                function_response=genai.protos.FunctionResponse(
                    name="get_weather",
                    response={"result": result}
                )
            )]
        )
    )

    print(response2.text)  # "서울의 날씨는 섭씨 22도로 맑습니다."

response = model.generate_content(
    """피보나치 수열의 20번째 항을 계산해주세요.
Python 코드를 작성하고 실행해서 결과를 알려주세요."""
)

print(response.text)

# 실행된 코드 확인
for part in response.candidates[0].content.parts:
    if hasattr(part, 'executable_code'):
        print("\n실행된 코드:")
        print(part.executable_code.code)
    if hasattr(part, 'code_execution_result'):
        print("\n실행 결과:")
        print(part.code_execution_result.output)

# 주의: Grounding은 Vertex AI에서만 지원 (Google Cloud)

from vertexai.preview.generative_models import (
    GenerativeModel,
    Tool,
    grounding
)

# Grounding 도구 설정

    grounding.GoogleSearchRetrieval()
)

model = GenerativeModel(
    "gemini-1.5-pro",
    tools=[google_search_tool]
)

# 최신 정보가 필요한 질문
response = model.generate_content(
    "2026년 1월 현재 세계 경제 상황은 어떤가요?"
)

print(response.text)

# 검색 출처 확인
if hasattr(response, 'grounding_metadata'):
    print("\n검색 출처:")
    for chunk in response.grounding_metadata.grounding_chunks:
        print(f"- {chunk.web.uri}")

모범 사례

권장사항 
// 1. 모델 선택
간단한 작업 → gemini-1.5-flash-8b (초저렴)
일반 작업 → gemini-1.5-flash
긴 컨텍스트/복잡한 작업 → gemini-1.5-pro

// 2. 컨텍스트 최적화
• 2M 토큰이 필요한 경우는 드물음
• 필요한 부분만 전달
• 요약 → 상세 분석 패턴 활용

// 3. 멀티모달 활용
• 이미지는 압축하여 전송
• 비디오는 긴 파일만 (짧은 파일은 프레임 추출)
• 파일 업로드 후 삭제로 스토리지 관리

// 4. 비용 절감
• 무료 티어 최대 활용 (프로토타이핑)
• 프롬프트 캐싱 (50% 할인)
• Flash-8b 모델 우선 시도

// 5. 에러 처리
try:
    response = model.generate_content(prompt)
except genai.types.StopCandidateException:
    print("안전 필터에 의해 차단됨")
except genai.types.BlockedPromptException:
    print("프롬프트가 차단됨")
except Exception as e:
    print(f"에러: {e}")

// 6. Rate Limit 대응
• 무료 티어: 분당 15/2 요청
• 유료 티어: 높은 한도
• 지수 백오프 재시도 구현
