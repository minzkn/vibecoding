export async function buildApp(): Promise<FastifyInstance> {

    logger: true,
  });

    title: string;
    completed: boolean;
  }
  let todos: Todo[] = [];

  // GET - ëª©ë¡ ì¡°íšŒ
  app.get('/todos', async (request, reply) => {

  // POST - ìƒì„± (Schema Validation)

      body: {
        type: 'object',
        required: ['title'],
        properties: {
          title: { type: 'string', minLength: 1 },
        },
      },
    },
  }, async (request, reply) => {
    const { title } = request.body as { title: string };
    const todo: Todo = { id: Date.now(), title, completed: false };
    todos.push(todo);
    reply.code(201);
    return todo;
  });

  // PUT - ìˆ˜ì •
  app.put('/todos/:id', async (request, reply) => {
    const { id } = request.params as { id: string };
    const { completed } = request.body as { completed: boolean };
    const index = todos.findIndex(t => t.id === Number(id));
    
    if (index === -1) {
      reply.code(404);
      return { error: "Not found" };
    }
    
    todos[index].completed = completed;
    return todos[index];
  });

  // DELETE - ì‚­ì œ
  app.delete('/todos/:id', async (request, reply) => {
    const { id } = request.params as { id: string };
    todos = todos.filter(t => t.id !== Number(id));
    reply.code(204);
  });

  return app;
}
